1. git clone https://github.com/InzynieriaOprogramowaniaAGH/MDO2022_S
	sklonowanie repozytorium z serwera na maszyne lokalną
	
2. git checkout -b AF303741
	utworzenie i przełączenie się na lokalnego branch'a o nazwie AF303741

3. utworzenie pliku sprawozdanie.txt w folderze AF303741/Lab01/sprawozdanie.txt

4. git status
	sprawdzenie statusu modyfikowanych/utworzonych/usuniętych plików
	
5. git add .
	dodanie modyfikowanych plików do śledzenia
	
6. git commit -m "[AF303741] Init commit"
	utworzenie commita z śledzonych plików
